% ---------------------------------------------------------------- float placement
% Every generated float is emitted [tb] (see cmp.report.table and book/build.py): top or bottom
% of a page, never "here". A float placed "here" lands wherever the paragraph happened to break,
% which strands it in mid-page with white space above and below; anchored to a top or bottom
% slot, the text closes over it and the page reads as one block.
%
% That only works if a slot is actually available, and LaTeX's defaults make sure it is not:
% \topfraction 0.7 means a float taller than 70 % of the text block can NEVER be a top float and
% is silently deferred to the end of the chapter — which is how a table ends up four pages from
% the paragraph that discusses it. These fractions buy the slots back.
\setcounter{topnumber}{2}              % up to 2 floats at the top of a page
\setcounter{bottomnumber}{1}           % and 1 at the bottom
\setcounter{totalnumber}{3}            % 3 floats on a page at most
\renewcommand{\topfraction}{0.9}       % a top float may take 90 % of the block (was 0.7)
\renewcommand{\bottomfraction}{0.5}    % a bottom float, half of it (was 0.3)
\renewcommand{\textfraction}{0.08}     % a page needs only 8 % text to not be a float page
\renewcommand{\floatpagefraction}{0.7} % a float page must be 70 % full, so it is not half-empty
\setlength{\floatsep}{14pt plus 3pt minus 2pt}       % between two floats
\setlength{\textfloatsep}{18pt plus 4pt minus 3pt}   % between a float and the text
\setlength{\intextsep}{14pt plus 3pt minus 2pt}
% Let a page end where its content ends, rather than stretching glue to force-justify the bottom.
% With many [tb] floats, \flushbottom (the book-class default) opens random gaps above/below floats
% and between paragraphs to make every page exactly full; \raggedbottom closes over the float and
% lets the leftover space fall at the very bottom of the page, where it reads as a natural page end.
\raggedbottom

% ---------------------------------------------------------------- the notebook pointer box
% Every chapter closes with one of these: where the runnable code is, which seed, FAST vs FULL.
% Deliberately a small note, not a section — the book is the artifact, the notebook is the lab.
\newcommand{\notebooknote}[1]{%
  % ragged right, not \sloppy: the note is dense with long unbreakable \code{} paths, and
  % justifying around them opens word gaps you can drive a bus through.

  \textcolor{SchemeGrey}{\rule{\linewidth}{0.4pt}}\par\smallskip

  \endgroup\medskip}
